import { Component, OnInit, Input, Output, EventEmitter } from '@angular/core';
import { FormGroup, FormBuilder, Validators } from '@angular/forms';
import { FormService } from 'src/app/services/form/form.service';

@Component({
  selector: 'app-dropdown',
  templateUrl: './dropdown.component.html',
  styleUrls: ['./dropdown.component.css']
})
export class DropdownComponent implements OnInit {

  @Input() parentform : FormGroup;
  @Input() name: string;
  @Input() labelname: string;
  @Input() required : boolean;
  @Input() submitted: boolean;
  @Input() options : any;
  @Input() optionvalue : any;
  @Input() showfield : any;
  @Input() displayfield : any;
  @Input() elementClass :any;
  @Output() valuechange = new EventEmitter();
  @Input() emitEvent : boolean;

  constructor(private fb: FormBuilder, private form : FormService) { }

  ngOnInit(): void {
    this.displayfield = this.displayfield ? this.displayfield : 'name';
    this.showfield = this.showfield !== undefined  ? this.showfield : true;
    this.elementClass = this.elementClass ? parseInt(this.elementClass) : 4;

    this.parentform.addControl(this.name ,this.fb.control(null, [ 
      this.form.conditionalValidator(this.required, Validators.required)]));

    if(this.emitEvent){
      this.parentform.controls[this.name].valueChanges.subscribe(() => { 
        this.valuechange.emit(this.parentform.controls[this.name].value);
      })
    }
  }

}

<div [formGroup]="parentform" class="row col-sm-12 nopadding nomargin" *ngIf="showfield">
  <label class="col-form-label col-sm-{{elementClass}}" for="{{name}}" >{{labelname}}
    <span class="required">*</span>
  </label>
  <div class="col-sm-6">
    <select formControlName="{{name}}" class="form-control">
      <option [ngValue]="null" disabled>Select {{labelname}}</option>
      <option *ngFor="let option of options" [ngValue]="option[optionvalue]">{{option[displayfield]}}</option>
    </select>
    <small class="text-danger"
        *ngIf="parentform.get(name).hasError('required') && ( parentform.get(name).touched || submitted)">
        Required Field
    </small>
  </div>
</div>
